（说明求解方法（解析解 / 数值解 / 启发式算法），给出算法流程，展示关键代码或伪代码。）

\begin{algorithm}[H]
    \caption{模型求解算法流程}
    \KwInput{输入数据 $D$，参数 $\theta$}
    \KwOutput{最优解 $x^*$}

    初始化 $x^{(0)}$\;
    \For{$k = 0, 1, 2, \dots, K_{\max}$}{
        计算梯度 $\nabla f(x^{(k)})$\;
        更新 $x^{(k+1)} = x^{(k)} - \eta \nabla f(x^{(k)})$\;
        \If{$\|x^{(k+1)} - x^{(k)}\| < \varepsilon$}{
            收敛，停止迭代\;
        }
    }
    输出 $x^* = x^{(k+1)}$\;
\end{algorithm}
